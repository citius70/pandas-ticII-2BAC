import pandas as pd

# Cargar dataset
url = "https://raw.githubusercontent.com/justmarkham/DAT8/master/data/winequality-red.csv"
df = pd.read_csv(url, sep=';')

# 游댲 Ejercicio 1: Conteo combinado por calidad y nivel de alcohol redondeado
df['alcohol_round'] = df['alcohol'].round()
print("游늵 Conteo combinado calidad vs alcohol:")
print(df.groupby(['quality', 'alcohol_round']).size().unstack())

df.groupby(['quality', 'alcohol_round']).size().unstack().T.plot(kind='bar', title='Calidad vs Nivel de Alcohol')

# 游댲 Ejercicio 2: Correlaci칩n de variables con la calidad
print("\n游댌 Correlaci칩n con la calidad:")
print(df.corr(numeric_only=True)['quality'].sort_values(ascending=False))

# 游댲 Ejercicio 3: Proporci칩n de vinos de alta calidad (>=7) por nivel de alcohol redondeado
df['high_quality'] = df['quality'] >= 7
print("\n游늳 Proporci칩n de alta calidad por alcohol:")
print(df.groupby('alcohol_round')['high_quality'].mean())

df.groupby('alcohol_round')['high_quality'].mean().plot(kind='bar', title='Proporci칩n de vinos de alta calidad por nivel de alcohol')

# 游댲 Ejercicio 4: Densidad promedio por calidad
print("\n游눦 Densidad promedio por calidad:")
print(df.groupby('quality')['density'].mean())

df.groupby('quality')['density'].mean().plot(kind='bar', title='Densidad promedio por calidad')

# 游댲 Ejercicio 5: Conteo de combinaciones 칰nicas de acidez vol치til y 치cido c칤trico
df['volatile acidity round'] = df['volatile acidity'].round(1)

tabla_combinaciones = df.groupby(['volatile acidity round', 'citric acid round']).size().unstack(fill_value=0)
print("\n游닍 Combinaciones de acidez vol치til y 치cido c칤trico (redondeadas):")
print(tabla_combinaciones.head())
